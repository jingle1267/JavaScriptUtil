### TypeUtil 工具里常用方法

```
npm install com.ihongqiqu.js.type
```

```javascript
    /**
     * 是否字符串
     * @param o
     * @returns {boolean}
     */
    isString(o)

    /**
     * 是否数字
     * @param o
     * @returns {boolean}
     */
    isNumber(o)

    /**
     * 是否boolean
     * @param o
     * @returns {boolean}
     */
    isBoolean(o)

    /**
     * 是否函数
     * @param o
     * @returns {boolean}
     */
    isFunction(o)

    /**
     * 是否为null
     * @param o
     * @returns {boolean}
     */
    isNull(o)

    /**
     * 是否undefined
     * @param o
     * @returns {boolean}
     */
    isUndefined(o)

    /**
     * 是否对象
     * @param o
     * @returns {boolean}
     */
    isObj(o)

    /**
     * 是否数组
     * @param o
     * @returns {boolean}
     */
    isArray(o)

    /**
     * 是否时间
     * @param o
     * @returns {boolean}
     */
    isDate(o)

    /**
     * 是否正则
     * @param o
     * @returns {boolean}
     */
    isRegExp(o)

    /**
     * 是否错误对象
     * @param o
     * @returns {boolean}
     */
    isError(o)

    /**
     * 是否Symbol函数
     * @param o
     * @returns {boolean}
     */
    isSymbol(o)

    /**
     * 是否Promise对象
     * @param o
     * @returns {boolean}
     */
    isPromise(o)

    /**
     * 是否Set对象
     * @param o
     * @returns {boolean}
     */
    isSet(o)

    /**
     * 是否 false
     * @param o
     * @returns {boolean}
     */
    isFalse(o)

    /**
     * 是否 true
     * @param o
     * @returns {boolean}
     */
    isTrue(o)
    
```
